/*
Desenvolva uma rotina que dado um número inteiro qualquer calcule seu antecessor
e seu sucessor.
*/
public class CalculoAntesEDepois {
    public static void main (String[] args) {
        int num = 20;
        int antes = num - 1;
        int depois = antes + 2;
        System.out.println("O número antecessor é " + antes + ", e o número sucessor é " + depois);
    }
}
